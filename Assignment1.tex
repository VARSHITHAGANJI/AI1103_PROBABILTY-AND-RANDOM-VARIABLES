\section*{Problem}
\textbf{Assigned Problem 6.9}
\\
If A and B are two events such that $A\subset B$ and $P(B)\neq 0$, then which of the following is correct?
\begin{enumerate}
   \item P$\brak {\frac{A}{B}}$ = $\frac{P\brak{B}}{P\brak{A}}$
   \item P$\brak {\frac{A}{B}}$ $< P\brak{A}$ 
   \item P$\brak {\frac{A}{B}}$ $\geq P\brak{A}$ 
   \item None of these
\end{enumerate}
\section*{Solution}
We know that A is the subset of B. \\
$\Rightarrow$ Every element of A is an element of B.\\
\begin{align}
  \therefore A\cap B = A
  \label{eq1}
\end{align}
We know that 
\begin{equation}
 \begin{split}
 P\brak{\frac{A}{B}}&= \frac{P\brak{A\cap B}}{P\brak{B}}  \\ 
 &=\frac{P\brak{A}}{P\brak{B}}
 \label{eq2}
\end{split}
\end{equation}
Given $0<P\brak{B}\leq1$
\begin{align}
  \Rightarrow \frac{1}{P\brak{B}} \geq 1 
\end{align}  

\textnormal{By multiplying with P$\brak{A}$ on both sides of the inequality, we get}\\
\\
\begin{align}
 \frac{P\brak{A}}{P\brak{B}}\geq {\normalsize{P\brak{A}}}
\end{align}
Using \ref{eq2}, we have 
$$P \brak {\frac{A}{B}} \geq P\brak{A}$$
Therefore, option 3 is correct.
